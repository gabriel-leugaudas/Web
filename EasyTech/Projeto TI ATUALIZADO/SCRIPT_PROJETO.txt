use projeto;
CREATE TABLE CLIENTE_CLI(
CLI_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
CLI_NOME_COMPLETO VARCHAR(50),
CLI_ENDERECO VARCHAR(100),
CLI_NUMERO VARCHAR(6),
CLI_COMPLEMENTO VARCHAR(50),
CLI_BAIRRO VARCHAR(50),
CLI_CEP VARCHAR(8),
CLI_UF VARCHAR (2),
CLI_CIDADE VARCHAR (25),
CLI_CPF VARCHAR(11),
CLI_EMAIL VARCHAR(50),
CLI_USUARIO VARCHAR(50),
CLI_SENHA VARCHAR(20)
);


CREATE TABLE TIPO_TELEFONE_TTE(
TTE_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
TTE_DESCRICAO VARCHAR(20)
);



CREATE TABLE TIPO_STATUS_TST(
TST_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
TST_DESCRICAO VARCHAR(50)
);


CREATE TABLE TIPO_SERVICO_TSE(
TSE_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
TSE_DESCRICAO VARCHAR (20)
);

CREATE TABLE SERVICO_SER(
SER_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
SER_NOME VARCHAR(50),
SER_VALOR double,
TSE_CODIGO INT,
FOREIGN KEY (TSE_CODIGO) REFERENCES tipo_status_tst(TST_CODIGO)
); 


CREATE TABLE STATUS_SERVICO_SSE(
SSE_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
SSE_DESCRICAO VARCHAR(50),
SSE_DATA_HORA_ENTRADA DATETIME,
SSE_DATA_HORA_SAIDA DATETIME,
TST_CODIGO INT, 
CLI_CODIGO INT,
FOREIGN KEY (TST_CODIGO) REFERENCES tipo_status_tst (TST_CODIGO),
FOREIGN KEY (CLI_CODIGO) REFERENCES cliente_cli (CLI_CODIGO)
);

CREATE TABLE TELEFONE_TEL(
TEL_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
TEL_NUMERO VARCHAR(11),
TTE_CODIGO INT,
CLI_CODIGO INT,
FOREIGN KEY (TTE_CODIGO) REFERENCES tipo_telefone_tte(TTE_CODIGO),
FOREIGN KEY (CLI_CODIGO) REFERENCES cliente_cli(CLI_CODIGO)
);


CREATE TABLE AGENDAMENTO_AGE(
AGE_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
AGE_VALOR DOUBLE,
AGE_DATA_HORA_ENTRADA DATETIME,
AGE_DATA_HORA_SAIDA DATETIME,
SER_CODIGO INT, 
SSE_CODIGO INT, 
CLI_CODIGO INT, 
FOREIGN KEY (SER_CODIGO) REFERENCES servico_ser(SER_CODIGO),
FOREIGN KEY (SSE_CODIGO) REFERENCES status_servico_sse(SSE_CODIGO),
FOREIGN KEY (CLI_CODIGO) REFERENCES cliente_cli(CLI_CODIGO)
);
